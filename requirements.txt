# Install Instructions:
# ====================
# 
# Basic Installation:
# pip install -r requirements.txt
#
# For GPU Support:
# pip install torch torchvision torchaudio --index-url https://download.pytorch.org/whl/cu118
# pip install faiss-gpu
#
# For Development:
# pip install -r requirements.txt
# pip install -e .
#
# For Production (minimal):
# Create requirements-prod.txt with only runtime dependencies
# pip install --no-dev -r requirements-prod.txt
#
# Docker Installation:
# FROM python:3.9-slim
# COPY requirements.txt .
# RUN pip install --no-cache-dir -r requirements.txt
#
# Version Constraints:
# ====================
# - Python >= 3.10 (required for FastAPI and modern typing)
# - CUDA >= 11.0 (for GPU acceleration)
# - Redis >= 6.0 (for advanced features)
# - FFmpeg (for video processing)
#
# System Dependencies (Ubuntu/Debian):
# ====================================
# apt-get update && apt-get install -y \
#     ffmpeg \
#     libsm6 \
#     libxext6 \
#     libfontconfig1 \
#     libxrender1 \
#     libgl1-mesa-glx \
#     tesseract-ocr \
#     tesseract-ocr-eng \
#     redis-server
#
# macOS (Homebrew):
# ==================
# brew install ffmpeg tesseract redis
#
# Performance Notes:
# ==================
# 1. Use faiss-gpu instead of faiss-cpu for large-scale vector search
# 2. Install PyTorch with CUDA support for GPU acceleration
# 3. Consider using TensorRT for model optimization in production
# 4. Use Redis Cluster for high-availability caching
# 5. Install Triton Inference Server for optimized model serving
